## Convert Birth Years to Integers
birth_years = []
parts = []
year = 0
for items in legislators:
    parts = items[2].split("-")
    try:
        year = int(parts[0])
    except Exception:
        pass
    if parts[0]=="":
        year=0
    birth_years.append(year)
    items.append(year)
